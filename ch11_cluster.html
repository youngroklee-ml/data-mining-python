11.2.2 Jaccard index

jaccard_sim = np.eye(len(df))
for i in range(len(df) - 1):
  for j in range(i + 1, len(df)):
    jaccard_sim[i, j] = jaccard_score(df.loc[i], df.loc[j])
    jaccard_sim[j, i] = jaccard_sim[i, j]
