import pandas as pd
import numpy as np
from sklearn.metrics import (
  rand_score, adjusted_rand_score, 
  silhouette_samples, silhouette_score, 
  calinski_harabasz_score
)

sol1 = np.array([1, 1, 2, 2, 2, 3, 3])
sol2 = np.array([1, 1, 2, 3, 2, 1, 3])
